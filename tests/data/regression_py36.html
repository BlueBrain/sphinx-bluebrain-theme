syntax_highlight = short

    Code tabs¶

    Tabs can be used to show multiple examples without excessive space. The code can be entered directly or included from a file.

    import sys

    require sys

    """This is a sample for the documentation."""


def double(x):
    return 2.0 * x


def halve(x):
    return 0.5 * x


def list_of_three(x):
    return [x] * 3
